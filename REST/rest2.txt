models.py 
 
from django.db import models 
class Feature(models.Model): 
    name = models.CharField(max_length=100) 
    icon = models.CharField(max_length=100)  # store icon class (like from Font Awesome) 
 
    def __str__(self): 
        return self.name 
 
class Product(models.Model): 
    name = models.CharField(max_length=100) 
    category = models.CharField(max_length=100) 
    image = models.ImageField(upload_to='products/') 
 
    def __str__(self): 
        return self.name 
 




views.py 
 
from django.shortcuts import render 
from .models import Feature, Product 
def home(request): 
    features = Feature.objects.all() 
    products = Product.objects.all() 
    context = { 
        'store_name': 'My General Store', 
        'store_aim': 'To provide quality goods at affordable prices, right to your doorstep!', 
        'features': features, 
        'products': products, 
        'contact': { 
            'phone': '+91-1234567890', 
            'email': 'contact@generalstore.com', 
            'address': '123 Main Street, YourCity' 
        } 
    } 
    return render(request, 'store/home.html', context)
